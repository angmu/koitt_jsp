getServerName() << local host
getServerPort() << 8181
getMethod() << post인지get인지 넘어오는 방식을 알아낼수있음

getRequestURL() << url을 확인
getRequestURI() << uri을 확인 (서버 포트 빼고 뒤의 주소

getQueryString() << 프로토콜을 확인
